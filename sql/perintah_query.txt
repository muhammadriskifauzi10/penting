"SELECT * FROM namatabel;" => Perintah untuk menampilkan semua daftar isi kolom
"SELECT namakolom1, namakolom2 from namatabel;" => Perintah untuk menampilkan data kolom tertentu di dalam tabel
"INSERT INTO namatabel (namakolom) VALUES (nilai);" => Perintah untuk memasukkan data kedalam tabel
"INSERT INTO namatabel (namakolom) VALUES (nilai), (nilai), (nilai);" 
							=> Perintah untuk memasukkan multi data kedalam tabel
"UPDATE namatabel SET namakolom = 'des 3' WHERE id = 1;" => Perintah untuk update data di dalam tabel
"UPDATE namatabel SET namakolom1 = 'nilai', namakolom2 = 'nilai' where id = 1;"
							=> Perintah untuk update multi data di dalam tabel
"DELETE FROM namatabel WHERE id = 1;" => Perintah untuk menghapus data dalam tabel
"TRUNCATE namatabel;" => Perintah untuk menghapus semua data tanpa menghapus tabelnya
"DROP table namatabel;" => Perintah untuk menghapus tabel
"SELECT * FROM namatabel WHERE namakolom = 'nilai';" Perintah untuk mengambil data berdasarkan kolom tertentu
"SELECT namakolom AS namakolomganti FROM namatabel; => Perintah untuk meng-aliaskan namakolom
"SELECT a.namakolom AS namakolomganti FROM namatabel AS a;" Perintah untuk meng-aliaskan namatabel

--- Operator perbandingan ---
=, <, >, (!=, <>), =>, <=
--- Operator perbandingan like ---
'%a(Akhirannya a)', 'b%(Awalannya b)', '%c%', 'NOT LIKE'

"SELECT * FROM namatabel WHERE namakolom LIKE '%a%';" => Perintah untuk mencari data
"SELECT * FROM namatabel WHERE namakolom IS NULL;" => Perintah untuk mencari data pada kolom yang berisi null
"SELECT * FROM namatabel WHERE namakolom IS NOT NULL;" 
								=> Perintah untuk mencari data pada kolom yang tidak berisi null
"SELECT * FROM namatabel WHERE namakolom BETWEEN 2000 AND 4000;" => Perintah untuk mencari data antara nilai 1
												sampai nilai 2 
"SELECT * FROM namatabel WHERE namakolom NOT BETWEEN 2000 AND 4000;" => Perintah untuk mencari data tidak diantara
												nilai 1 dan nilai 2
"SELECT * FROM namatabel WHERE namakolom IN ('n1','n2');" => Perintah untuk mengambil data berdasarkan nilai
"SELECT * FROM namatabel WHERE namakolom NOT IN ('n1','n2');" => Perintah untuk mengambil data berdasarkan nilai
"SELECT * FROM namatabel ORDER BY namakolom ASC/DESC;" => Perintah untuk mengurutkan data dari urutan nilai yang terbesar
									dan terkecil
"SELECT * FROM namatabel ORDER BY namakolom DESC LIMIT 2;" => Perintah untuk mengambil jumlah data
"SELECT * FROM namatabel ORDER BY namakolom DESC LIMIT skip, limit;" => Perintah untuk mengambil jumlah data 
											berdasarkan skip urut data
"SELECT DISTINCT namakolom FROM namatabel;" => Perintah untuk mengambil data yang duplicat
"SELECT LAST_INSERT_ID();" perintah untuk menampilkan urutan masukan data 1 yang baru saja di query
